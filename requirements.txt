#    venv\Scripts\activate     (Windows)

# 2. Install dependencies:
#    pip install -r requirements.txt
#
# 3. For macOS system Python (if not using venv):
#    pip install -r requirements.txt --break-system-packages
#
# 4. Install system tools:
#    macOS:  brew install exiftool dcraw
#    Ubuntu: sudo apt-get install exiftool dcraw
#    Fedora: sudo dnf install perl-Image-ExifTool dcraw
